CREATE TABLE QUIZ(
QuizID CHAR(5) NOT NULL,
QuizQ CHAR(100),
QuizA CHAR(100),
Value INT,
Level INT,
CONSTRAINT PK_QUIZ PRIMARY KEY(QuizID));


CREATE TABLE HINTs(
HintID CHAR(5) NOT NULL,
Hint CHAR(100),
CONSTRAINT PK_HINTS PRIMARY KEY(HintID));

CREATE TABLE SCOREBOARD(
TeamID CHAR(5) NOT NULL,
Score INT,
Names CHAR(50),
CONSTRAINT PK_SCOREBOARD PRIMARY KEY(TeamID));
